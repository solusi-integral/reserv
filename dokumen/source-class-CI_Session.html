 51:     /**
 52:      * Session Constructor
 53:      *
 54:      * The constructor runs the session routines automatically
 55:      * whenever the class is instantiated.
 56:      */
 57:     public function __construct($params = array())
 58:     {
 59:         log_message('debug', "Session Class Initialized");
 60: 
 61:         // Set the super object to a local variable for use throughout the class
 62:         $this->CI =& get_instance();
 63: 
 64:         // Set all the session preferences, which can either be set
 65:         // manually via the $params array above or via the config file
 66:         foreach (array('sess_encrypt_cookie', 'sess_use_database', 'sess_table_name', 'sess_expiration', 'sess_expire_on_close', 'sess_match_ip', 'sess_match_useragent', 'sess_cookie_name', 'cookie_path', 'cookie_domain', 'cookie_secure', 'sess_time_to_update', 'time_reference', 'cookie_prefix', 'encryption_key') as $key)
 67:         {
 68:             $this->$key = (isset($params[$key])) ? $params[$key] : $this->CI->config->item($key);
 69:         }
 70: 
 71:         if ($this->encryption_key == '')
 72:         {
 73:             show_error('In order to use the Session class you are required to set an encryption key in your config file.');
 74:         }
 75: 
 76:         // Load the string helper so we can use the strip_slashes() function
 77:         $this->CI->load->helper('string');
 78: 
 79:         // Do we need encryption? If so, load the encryption class
 80:         if ($this->sess_encrypt_cookie == TRUE)
 81:         {
 82:             $this->CI->load->library('encrypt');
 83:         }
 84: 
 85:         // Are we using a database?  If so, load it
 86:         if ($this->sess_use_database === TRUE AND $this->sess_table_name != '')
 87:         {
 88:             $this->CI->load->database();
 89:         }
 90: 
 91:         // Set the "now" time.  Can either be GMT or server time, based on the
 92:         // config prefs.  We use this to set the "last activity" time
 93:         $this->now = $this->_get_time();
 94: 
 95:         // Set the session length. If the session expiration is
 96:         // set to zero we'll set the expiration two years from now.
 97:         if ($this->sess_expiration == 0)
 98:         {
 99:             $this->sess_expiration = (60*60*24*365*2);
100:         }
101: 
102:         // Set the cookie name
103:         $this->sess_cookie_name = $this->cookie_prefix.$this->sess_cookie_name;
104: 
105:         // Run the Session routine. If a session doesn't exist we'll
106:         // create a new one.  If it does, we'll update it.
107:         if ( ! $this->sess_read())
108:         {
109:             $this->sess_create();
110:         }
111:         else
112:         {
113:             $this->sess_update();
114:         }
115: 
116:         // Delete 'old' flashdata (from last request)
117:         $this->_flashdata_sweep();
118: 
119:         // Mark all new flashdata as old (data will be deleted before next request)
120:         $this->_flashdata_mark();

130:     /**
131:      * Fetch the current session data if it exists
132:      *
133:      * @access  public
134:      * @return  bool
135:      */
136:     function sess_read()
137:     {
138:         // Fetch the cookie
139:         $session = $this->CI->input->cookie($this->sess_cookie_name);
140: 
141:         // No cookie?  Goodbye cruel world!...
142:         if ($session === FALSE)
143:         {
144:             log_message('debug', 'A session cookie was not found.');
145:             return FALSE;
146:         }
147: 
148:         // HMAC authentication
149:         $len = strlen($session) - 40;
150: 
151:         if ($len <= 0)
152:         {
153:             log_message('error', 'Session: The session cookie was not signed.');
154:             return FALSE;
155:         }
156: 
157:         // Check cookie authentication
158:         $hmac = substr($session, $len);
159:         $session = substr($session, 0, $len);
160: 
161:         // Time-attack-safe comparison
162:         $hmac_check = hash_hmac('sha1', $session, $this->encryption_key);
163:         $diff = 0;
164: 
165:         for ($i = 0; $i < 40; $i++)
166:         {
167:             $xor = ord($hmac[$i]) ^ ord($hmac_check[$i]);
168:             $diff |= $xor;
169:         }
170: 
171:         if ($diff !== 0)
172:         {
173:             log_message('error', 'Session: HMAC mismatch. The session cookie data did not match what was expected.');
174:             $this->sess_destroy();
175:             return FALSE;
176:         }
177: 
178:         // Decrypt the cookie data
179:         if ($this->sess_encrypt_cookie == TRUE)
180:         {
181:             $session = $this->CI->encrypt->decode($session);
182:         }
183: 
184:         // Unserialize the session array
185:         $session = $this->_unserialize($session);
186: 
187:         // Is the session data we unserialized an array with the correct format?
188:         if ( ! is_array($session) OR ! isset($session['session_id']) OR ! isset($session['ip_address']) OR ! isset($session['user_agent']) OR ! isset($session['last_activity']))
189:         {
190:             $this->sess_destroy();
191:             return FALSE;
192:         }
193: 
194:         // Is the session current?
195:         if (($session['last_activity'] + $this->sess_expiration) < $this->now)
196:         {
197:             $this->sess_destroy();
198:             return FALSE;
199:         }
200: 
201:         // Does the IP Match?
202:         if ($this->sess_match_ip == TRUE AND $session['ip_address'] != $this->CI->input->ip_address())
203:         {
204:             $this->sess_destroy();
205:             return FALSE;
206:         }
207: 
208:         // Does the User Agent Match?
209:         if ($this->sess_match_useragent == TRUE AND trim($session['user_agent']) != trim(substr($this->CI->input->user_agent(), 0, 120)))
210:         {
211:             $this->sess_destroy();
212:             return FALSE;
213:         }
214: 
215:         // Is there a corresponding session in the DB?
216:         if ($this->sess_use_database === TRUE)
217:         {
218:             $this->CI->db->where('session_id', $session['session_id']);
219: 
220:             if ($this->sess_match_ip == TRUE)
221:             {
222:                 $this->CI->db->where('ip_address', $session['ip_address']);
223:             }
224: 
225:             if ($this->sess_match_useragent == TRUE)
226:             {
227:                 $this->CI->db->where('user_agent', $session['user_agent']);
228:             }
229: 
230:             $query = $this->CI->db->get($this->sess_table_name);
231: 
232:             // No result?  Kill it!
233:             if ($query->num_rows() == 0)
234:             {
235:                 $this->sess_destroy();
236:                 return FALSE;
237:             }
238: 
239:             // Is there custom data?  If so, add it to the main session array
240:             $row = $query->row();
241:             if (isset($row->user_data) AND $row->user_data != '')
242:             {
243:                 $custom_data = $this->_unserialize($row->user_data);
244: 
245:                 if (is_array($custom_data))
246:                 {
247:                     foreach ($custom_data as $key => $val)
248:                     {
249:                         $session[$key] = $val;
250:                     }
251:                 }
252:             }
253:         }
254: 
255:         // Session is valid!
256:         $this->userdata = $session;
257:         unset($session);
258: 
259:         return TRUE;
260:     }

264:     /**
265:      * Write the session data
266:      *
267:      * @access  public
268:      * @return  void
269:      */
270:     function sess_write()
271:     {
272:         // Are we saving custom data to the DB?  If not, all we do is update the cookie
273:         if ($this->sess_use_database === FALSE)
274:         {
275:             $this->_set_cookie();
276:             return;
277:         }
278: 
279:         // set the custom userdata, the session data we will set in a second
280:         $custom_userdata = $this->userdata;
281:         $cookie_userdata = array();
282: 
283:         // Before continuing, we need to determine if there is any custom data to deal with.
284:         // Let's determine this by removing the default indexes to see if there's anything left in the array
285:         // and set the session data while we're at it
286:         foreach (array('session_id','ip_address','user_agent','last_activity') as $val)
287:         {
288:             unset($custom_userdata[$val]);
289:             $cookie_userdata[$val] = $this->userdata[$val];
290:         }
291: 
292:         // Did we find any custom data?  If not, we turn the empty array into a string
293:         // since there's no reason to serialize and store an empty array in the DB
294:         if (count($custom_userdata) === 0)
295:         {
296:             $custom_userdata = '';
297:         }
298:         else
299:         {
300:             // Serialize the custom data array so we can store it
301:             $custom_userdata = $this->_serialize($custom_userdata);
302:         }
303: 
304:         // Run the update query
305:         $this->CI->db->where('session_id', $this->userdata['session_id']);
306:         $this->CI->db->update($this->sess_table_name, array('last_activity' => $this->userdata['last_activity'], 'user_data' => $custom_userdata));
307: 
308:         // Write the cookie.  Notice that we manually pass the cookie data array to the
309:         // _set_cookie() function. Normally that function will store $this->userdata, but
310:         // in this case that array contains custom data, which we do not want in the cookie.
311:         $this->_set_cookie($cookie_userdata);
312:     }

323:     {

326:         {
327:             $sessid .= mt_rand(0, mt_getrandmax());
328:         }

330:         // To make the session ID even more secure we'll combine it with the user's IP
331:         $sessid .= $this->CI->input->ip_address();

334:                             'session_id'    => md5(uniqid($sessid, TRUE)),
335:                             'ip_address'    => $this->CI->input->ip_address(),
336:                             'user_agent'    => substr($this->CI->input->user_agent(), 0, 120),
337:                             'last_activity' => $this->now,
338:                             'user_data'     => ''
339:                             );
340:

342:         // Save the data to the DB if needed
343:         if ($this->sess_use_database === TRUE)
344:         {
345:             $this->CI->db->query($this->CI->db->insert_string($this->sess_table_name, $this->userdata));
346:         }
347: 
348:         // Write the cookie
349:         $this->_set_cookie();

354:     /**
355:      * Update an existing session
356:      *
357:      * @access  public
358:      * @return  void
359:      */
360:     function sess_update()
361:     {
362:         // We only update the session every five minutes by default
363:         if ($this->CI->input->is_ajax_request() OR ($this->userdata['last_activity'] + $this->sess_time_to_update) >= $this->now)
364:         {
365:             return;
366:         }
367: 
368:         // Save the old session id so we know which record to
369:         // update in the database if we need it
370:         $old_sessid = $this->userdata['session_id'];
371:         $new_sessid = '';
372:         while (strlen($new_sessid) < 32)
373:         {
374:             $new_sessid .= mt_rand(0, mt_getrandmax());
375:         }
376: 
377:         // To make the session ID even more secure we'll combine it with the user's IP
378:         $new_sessid .= $this->CI->input->ip_address();
379: 
380:         // Turn it into a hash
381:         $new_sessid = md5(uniqid($new_sessid, TRUE));
382: 
383:         // Update the session data in the session data array
384:         $this->userdata['session_id'] = $new_sessid;
385:         $this->userdata['last_activity'] = $this->now;
386: 
387:         // _set_cookie() will handle this for us if we aren't using database sessions
388:         // by pushing all userdata to the cookie.
389:         $cookie_data = NULL;
390: 
391:         // Update the session ID and last_activity field in the DB if needed
392:         if ($this->sess_use_database === TRUE)
393:         {
394:             // set cookie explicitly to only have our session data
395:             $cookie_data = array();
396:             foreach (array('session_id','ip_address','user_agent','last_activity') as $val)
397:             {
398:                 $cookie_data[$val] = $this->userdata[$val];
399:             }
400: 
401:             $this->CI->db->query($this->CI->db->update_string($this->sess_table_name, array('last_activity' => $this->now, 'session_id' => $new_sessid), array('session_id' => $old_sessid)));
402:         }
403: 
404:         // Write the cookie
405:         $this->_set_cookie($cookie_data);
406:     }

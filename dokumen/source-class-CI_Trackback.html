 66:         // Pre-process the Trackback Data

 75:             switch ($item)
 76:             {
 77:                 case 'ping_url' : $$item = $this->extract_urls($tb_data[$item]);

221:     /**
222:      * Process Trackback
223:      *
224:      * Opens a socket connection and passes the data to
225:      * the server.  Returns TRUE on success, FALSE on failure
226:      *
227:      * @access  public
228:      * @param   string
229:      * @param   string
230:      * @return  bool
231:      */
232:     function process($url, $data)
233:     {
234:         $target = parse_url($url);
235: 
236:         // Open the socket
237:         if ( ! $fp = @fsockopen($target['host'], 80))
238:         {
239:             $this->set_error('Invalid Connection: '.$url);
240:             return FALSE;
241:         }
242: 
243:         // Build the path
244:         $ppath = ( ! isset($target['path'])) ? $url : $target['path'];
245: 
246:         $path = (isset($target['query']) && $target['query'] != "") ? $ppath.'?'.$target['query'] : $ppath;
247: 
248:         // Add the Trackback ID to the data string
249:         if ($id = $this->get_id($url))
250:         {
251:             $data = "tb_id=".$id."&".$data;
252:         }
253: 
254:         // Transfer the data
255:         fputs ($fp, "POST " . $path . " HTTP/1.0\r\n" );
256:         fputs ($fp, "Host: " . $target['host'] . "\r\n" );
257:         fputs ($fp, "Content-type: application/x-www-form-urlencoded\r\n" );
258:         fputs ($fp, "Content-length: " . strlen($data) . "\r\n" );
259:         fputs ($fp, "Connection: close\r\n\r\n" );
260:         fputs ($fp, $data);
261: 
262:         // Was it successful?
263:         $this->response = "";
264: 
265:         while ( ! feof($fp))
266:         {
267:             $this->response .= fgets($fp, 128);
268:         }
269:         @fclose($fp);
270: 
271: 
272:         if (stristr($this->response, '<error>0</error>') === FALSE)
273:         {
274:             $message = 'An unknown error was encountered';
275: 
276:             if (preg_match("/<message>(.*?)<\/message>/is", $this->response, $match))
277:             {
278:                 $message = trim($match['1']);
279:             }
280: 
281:             $this->set_error($message);
282:             return FALSE;
283:         }

285:         return TRUE;

304:         $urls = preg_replace("/\s*(\S+)\s*/", "\\1,", $urls);

306:         // If they use commas get rid of the doubles.
307:         $urls = str_replace(",,", ",", $urls);

310:         if (substr($urls, -1) == ",")
311:         {

425:      * Character limiter
426:      *
427:      * Limits the string based on the character count. Will preserve complete words.
428:      *
429:      * @access  public
430:      * @param   string
431:      * @param   integer
432:      * @param   string
433:      * @return  string
434:      */
435:     function limit_characters($str, $n = 500, $end_char = '&#8230;')
436:     {
437:         if (strlen($str) < $n)
438:         {
439:             return $str;
440:         }
441: 
442:         $str = preg_replace("/\s+/", ' ', str_replace(array("\r\n", "\r", "\n"), ' ', $str));
443: 
444:         if (strlen($str) <= $n)
445:         {
446:             return $str;
447:         }
448: 
449:         $out = "";
450:         foreach (explode(' ', trim($str)) as $val)
451:         {
452:             $out .= $val.' ';
453:             if (strlen($out) >= $n)
454:             {
455:                 return trim($out).$end_char;
456:             }
457:         }
458:     }

463:      * High ASCII to Entities
464:      *
465:      * Converts Hight ascii text and MS Word special chars
466:      * to character entities
467:      *
468:      * @access  public
469:      * @param   string
470:      * @return  string
471:      */
472:     function convert_ascii($str)
473:     {
474:         $count  = 1;
475:         $out    = '';
476:         $temp   = array();
477: 
478:         for ($i = 0, $s = strlen($str); $i < $s; $i++)
479:         {
480:             $ordinal = ord($str[$i]);
481: 
482:             if ($ordinal < 128)
483:             {
484:                 $out .= $str[$i];
485:             }
486:             else
487:             {
488:                 if (count($temp) == 0)
489:                 {
490:                     $count = ($ordinal < 224) ? 2 : 3;
491:                 }
492: 
493:                 $temp[] = $ordinal;
494: 
495:                 if (count($temp) == $count)
496:                 {
497:                     $number = ($count == 3) ? (($temp['0'] % 16) * 4096) + (($temp['1'] % 64) * 64) + ($temp['2'] % 64) : (($temp['0'] % 32) * 64) + ($temp['1'] % 64);
498: 
499:                     $out .= '&#'.$number.';';
500:                     $count = 1;
501:                     $temp = array();
502:                 }
503:             }
504:         }
505: 
506:         return $out;
507:     }
